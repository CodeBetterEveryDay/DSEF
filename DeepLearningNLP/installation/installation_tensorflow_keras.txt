## You can install TensorFlow in different ways:

# 1 Python and Virtualenv or Conda: In this approach, you install TensorFlow and Keras and all of the packages in a Python virtual environment or conda environment.

# Conda - Ubuntu 16.04 P3.6 +

sudo apt-get update

conda create -n tf-keras-conda-p36 --copy python=3.6
source activate tf-keras-conda-p36        source deactivate     atom ~/.keras/keras.json

conda install jupyter

conda install scipy
pip install sklearn
pip install pandas

pip install tensorflow
pip install keras

pip install --upgrade tensorflow
pip install --upgrade keras

## you can include the above and install with requirements.txt
pip install -r requirements.txt
